### Parameters
<ParamField path="itemId" type="string" required>
  This is the Id of the item.
</ParamField>

### Response

<ResponseField name="itemName" type="strig">
  Name of Item
</ResponseField>

<ResponseField name="ownerId" type="string">
  Owner Id of Item
</ResponseField>

<ResponseField name="characteristics" type="object">
  Characterstics of Item with Blockchain Id
<Expandable title="Toggle object">

<ResponseField name="caracteristicOne" type="string">
  Item Characterstic One
</ResponseField>

<ResponseField name="caracteristicTwo" type="string">
  Item Characterstic Two
</ResponseField>

<ResponseField name="caracteristicThree" type="string">
  Item Characterstic Three
</ResponseField>

<ResponseField name="blockchainId" type="string">
  Id from Blockchain
</ResponseField>

</Expandable>

<ResponseField name="itemId" type="string">
  Id of the Item
</ResponseField>

</ResponseField>
