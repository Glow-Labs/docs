---
title: "Update Item"
api: "PUT https://game-server-dev.fly.dev/api/v1/items/{itemId}"
description: "This endpoint updates an existing item."
---

### Parameters
<ParamField header="Authorization" type="string" required>
  This is the Authorization token of customer.
</ParamField>

### Body

<ParamField path="itemId" type="string" required>
  This is the Id of the item.
</ParamField>

<ParamField body="itemName" type="string">
  This is Id of the Name of the Item.
</ParamField>

<ParamField body="ownerId" type="string">
  This is Id of the Owner of the Item.
</ParamField>

<ParamField body="characteristics" type="object">
  This object contains specific characteristics related to the item.
    <ParamField body="characteristics.characteristic.One" type="string">
      Identifier or value associated with `characteristic.One` of the item.
    </ParamField>
    <ParamField body="characteristics.characteristic.Two" type="string">
      Identifier or value associated with `characteristic.Two` of the item.
    </ParamField>

    <ParamField body="characteristics.characteristic.Three" type="string">
      Identifier or value associated with `characteristic.Three` of the item.
    </ParamField>

    <ParamField body="characteristics.blockchainId" type="string">
      The blockchain ID related to the item, indicating its unique identifier on the blockchain.
    </ParamField>
</ParamField>
