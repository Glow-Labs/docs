---
title: "Get Event"
api: "GET https://game-server-dev.fly.dev/api/v1/event"
description: "This endpoint get an Event."
---

### Parameters
<ParamField header="Authorization" type="string" required>
  This is the Authorization token of customer.
</ParamField>

### Body
<ParamField query="questId" type="string">
  This is the Id of the Quest
</ParamField>

<ParamField query="userId" type="string">
  This is the Id of the User
</ParamField>

<ParamField query="itemId" type="string">
  This is the Id of the Item
</ParamField>

<ParamField query="eventId" type="string">
 This is Id of the Event.
</ParamField>

### Response

<ResponseField name="events" type="array" required>
 An array containing information about each event.
</ResponseField>
<ResponseField name="events.userId" type="string" required>
 The ID of the user associated with the event.
</ResponseField>
<ResponseField name="events.customerId" type="string" required>
 The ID of the customer associated with the event.
</ResponseField>
<ResponseField name="events.questId" type="string" required>
 The ID of the quest associated with the event.
</ResponseField>
<ResponseField name="events.status" type="string" required>
 The status of the event.
</ResponseField>
<ResponseField name="events.itemId" type="string" required>
 The ID of the item associated with the event.
</ResponseField>
<ResponseField name="events.points" type="number" required>
 The points earned from the event.
</ResponseField>
<ResponseField name="events.eventId" type="string" required>
 The ID of the event.
</ResponseField>
<ResponseField name="events.lastClaimedAt" type="string" required>
 The timestamp when the event was last claimed (ISO 8601 date format).
</ResponseField>
<ResponseField name="totalEvents" type="number" required>
 Total number of events.
</ResponseField>
<ResponseField name="limit" type="number" required>
 Limit of events per page.
</ResponseField>
<ResponseField name="page" type="number" required>
 Current page number.
</ResponseField>
<ResponseField name="totalPages" type="number" required>
 Total number of pages.
</ResponseField>
<ResponseField name="pagingCounter" type="number" required>
 Counter for pagination.
</ResponseField>
<ResponseField name="hasPrevPage" type="boolean" required>
 Indicates if there is a previous page.
</ResponseField>
<ResponseField name="hasNextPage" type="boolean" required>
 Indicates if there is a next page.
</ResponseField>
<ResponseField name="prevPage" type="number|null" required>
 Number of the previous page, or null if not available.
</ResponseField>
<ResponseField name="nextPage" type="number|null" required>
 Number of the next page, or null if not available.
</ResponseField>
