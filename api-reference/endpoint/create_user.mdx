---
title: "Create User"
api: "POST https://game-server-dev.fly.dev/api/v1/user"
description: "This endpoint creates a new user"
---

### Body
<ParamField header="Authorization" type="string" required>
  This is the Authorization token of customer.
</ParamField>
<ParamField body="discordId" type="string" required>
  This is discord Id of the User
</ParamField>
<ParamField body="discordEmail" type="string" required>
  This is discord Email of the User
</ParamField>
<ParamField body="name" type="string" required>
  This is the name of the user
</ParamField>
<ParamField body="customerId" type="string">
  This is discord Id of the User
</ParamField>

### Response

<ResponseField name="userId" type="string" required>
  This is the Id of the User
</ResponseField>

<ResponseField name="discordId" type="string" required>
  This is discord Id of the User
</ResponseField>

<ResponseField name="discordEmail" type="string" required>
  This is discord Email of the User
</ResponseField>

<ResponseField name="name" type="string" required>
  This is the name of the user
</ResponseField>
